import mysql.connector
from flask import Flask, render_template
from zoneinfo import ZoneInfo  
from datetime import datetime


app = Flask(__name__)

@app.route('/')
def home():
    # Connect to MariaDB
    conn = mysql.connector.connect(
        host="localhost",
        user="exampleuser",
        password="change_this_strong_password",
        database="exampledb"
    )
    cursor = conn.cursor()
    cursor.execute("SELECT NOW()")
    result = cursor.fetchone()
    # Clean up
    cursor.close()
    conn.close()

    utc_time = result[0].replace(tzinfo=ZoneInfo("UTC"))
    local_time = utc_time.astimezone(ZoneInfo("Europe/Helsinki"))

    return render_template('index.html', time=local_time.strftime('%Y-%m-%d %H:%M:%S'))

if __name__ == '__main__':
    app.run(host='0.0.0.0', port=5000)
